def <=>(other)
   mcm = (@denominador * other.denominador)/gcd(@denominador, other.denominador)
   a = (mcm/@denominador*@numerador)
   b = (mcm/other.denominador*other.numerador)

   if a < b
      -1
   elsif a > b
      1
   else
      0
   end
end
